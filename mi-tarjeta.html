1Abre tu Google Sheet
Ve a tu hoja Mi Tarjeta - Consumos → menú Extensiones → Apps Script
2Pega este código y guarda
Borra todo lo que hay en el editor y pega esto:
function doGet(e){ return handle(e); } function doPost(e){ return handle(e); } function handle(e){ var ss = SpreadsheetApp .openById("1b8gOPtgHq0Wg1g-kRvOrHUbrPcf7aaBqL_Co6pl8uU4"); var sheet = ss.getSheets()[0]; var p = e.parameter; var out = {}; if(p.action === "read"){ var rows = sheet.getDataRange().getValues(); out = {txs: rows.slice(1).map(function(r){ return {date:r[0],amount:r[1],catIdx:r[2],desc:r[3]}; }), limit: Number(sheet.getRange("F1").getValue())||1000}; } else if(p.action === "add"){ var d = JSON.parse(p.data); sheet.appendRow([d.date,d.amount,d.catIdx,d.desc]); out = {ok:true}; } else if(p.action === "delete"){ var rows2 = sheet.getDataRange().getValues(); for(var i=rows2.length-1;i>=1;i--){ if(String(rows2[i][0])===String(p.date)){ sheet.deleteRow(i+1); break; } } out = {ok:true}; } else if(p.action === "setLimit"){ sheet.getRange("F1").setValue(Number(p.limit)); out = {ok:true}; } else if(p.action === "reset"){ var last = sheet.getLastRow(); if(last > 1) sheet.deleteRows(2, last-1); out = {ok:true}; } return ContentService .createTextOutput(JSON.stringify(out)) .setMimeType(ContentService.MimeType.JSON); }
